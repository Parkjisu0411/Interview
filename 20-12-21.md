```java
class DBox<T1, T2> {
	private T1 ob1;
	private T2 ob2;
	
	public void set(T1 o1, T2 o2) {
		ob1 = o1;
		ob2 = o2;
	}
	
	@Override
	public String toString() {
		return ob1 + " & " + ob2;
	}
}

class DDBox<T1, T2> {
	private T1 ob1;
	private T2 ob2;
	
	public void set(T1 o1, T2 o2) {
		ob1 = o1;
		ob2 = o2;
	}
	
	@Override
	public String toString() {
		return ob1 + "\n" + ob2;
	}
}
```

```java
public class Main {
	public static void main(String[] args) {
		Box<Integer> box1 = new Box<>();
        box1.set(99);

        Box<String> box2 = new Box<>();
        box2.set("55");
        
        System.out.println(box1.get() + " & " + box2.get());
        swapBox(box1, box2);
        System.out.println(box1.get() + " & " + box2.get());
        
        //99 & 55
        //55 & 99
	}
	
	public static void swapBox(Box box1, Box box2 ) {
		Box temp = new Box();
		temp.set(box1.get());
		box1.set(box2.get());
		box2.set(temp.get());
	}
	
}

class Box<T> {
	private T ob;
	
	public void set(T o) {
		ob = o;
	}
	
	public T get() {
		return ob;
	}
}
```
